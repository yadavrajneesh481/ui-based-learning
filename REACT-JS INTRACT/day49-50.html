Single File Upload

// HTML <input type="file" id="fileInput" accept="image/*" /> <button onclick="uploadFile()">Upload</button> // JavaScript function uploadFile() { const fileInput = document.getElementById('fileInput'); const file = fileInput.files[0]; if (!file) { alert('Please select a file'); return; } // File validation const maxSize = 5 * 1024 * 1024; // 5MB if (file.size > maxSize) { alert('File size should be less than 5MB'); return; } // Create FormData const formData = new FormData(); formData.append('file', file); // Upload to server fetch('/api/upload', { method: 'POST', body: formData }) .then(response => response.json()) .then(data => { console.log('Upload successful:', data); }) .catch(error => { console.error('Upload failed:', error); }); }

Multiple File Upload

// HTML <input type="file" id="multiFileInput" multiple accept="image/*" /> // JavaScript function uploadMultipleFiles() { const fileInput = document.getElementById('multiFileInput'); const files = Array.from(fileInput.files); if (files.length === 0) { alert('Please select files'); return; } const formData = new FormData(); // Append all files files.forEach((file, index) => { formData.append(`file${index}`, file); }); // Upload with progress tracking const xhr = new XMLHttpRequest(); xhr.upload.addEventListener('progress', (e) => { if (e.lengthComputable) { const percentComplete = (e.loaded / e.total) * 100; console.log(`Upload progress: ${percentComplete}%`); } }); xhr.addEventListener('load', () => { if (xhr.status === 200) { console.log('Upload successful'); } }); xhr.open('POST', '/api/upload-multiple'); xhr.send(formData); }

‚úÖ Client-Side Validation Architecture

Client-side validation user experience improve karta hai aur unnecessary server requests reduce karta hai. Lekin yaad rakhein, server-side validation bhi mandatory hai security ke liye.

Form Validation

Client-side validation is the process of checking user input in the browser before sending it to the server. It provides immediate feedback and reduces server load.

Validation Types

Validation Type Description Example
Required Field must not be empty Name, Email
Format Input must match pattern Email, Phone, URL
Length Min/Max character limit Password (8-20 chars)
Range Numeric value range Age (18-100)
Custom Business logic validation Password strength

Validation Implementation

// Validation Rules const validationRules = { email: { required: true, pattern: /^[^\s@]+@[^\s@]+\.[^\s@]+$/, message: 'Please enter a valid email' }, password: { required: true, minLength: 8, pattern: /^(?=.*[a-z])(?=.*[A-Z])(?=.*\d)(?=.*[@$!%*?&])/, message: 'Password must contain uppercase, lowercase, number and special character' }, phone: { required: true, pattern: /^\d{10}$/, message: 'Please enter a valid 10-digit phone number' } }; // Validation Function function validateField(fieldName, value) { const rules = validationRules[fieldName]; if (!rules) return { valid: true }; // Required check if (rules.required && !value.trim()) { return { valid: false, message: `${fieldName} is required` }; } // Pattern check if (rules.pattern && !rules.pattern.test(value)) { return { valid: false, message: rules.message }; } // Length check if (rules.minLength && value.length < rules.minLength) { return { valid: false, message: `${fieldName} must be at least ${rules.minLength} characters` }; } return { valid: true }; } // Form Validation function validateForm(formData) { const errors = {}; Object.keys(formData).forEach(field => { const result = validateField(field, formData[field]); if (!result.valid) { errors[field] = result.message; } }); return { valid: Object.keys(errors).length === 0, errors }; }

üö® Error Handling & Error Boundaries

Error handling ek robust application ka essential part hai. Proper error handling se application crash nahi hota aur user ko clear feedback milta hai.

Error Boundary

Error Boundary is a React component that catches JavaScript errors anywhere in its child component tree, logs those errors, and displays a fallback UI instead of crashing the entire application.

Error Boundary Implementation

import React from 'react'; class ErrorBoundary extends React.Component { constructor(props) { super(props); this.state = { hasError: false, error: null, errorInfo: null }; } static getDerivedStateFromError(error) { // Update state so next render shows fallback UI return { hasError: true }; } componentDidCatch(error, errorInfo) { // Log error to error reporting service console.error('Error caught by boundary:', error, errorInfo); this.setState({ error: error, errorInfo: errorInfo }); // Send to error tracking service (e.g., Sentry) // logErrorToService(error, errorInfo); } render() { if (this.state.hasError) { // Fallback UI return ( <div className="error-boundary"> <h2>üòî Something went wrong</h2> <p>We're sorry for the inconvenience. Please try refreshing the page.</p> {process.env.NODE_ENV === 'development' && ( <details style={{ whiteSpace: 'pre-wrap' }}> <summary>Error Details</summary> {this.state.error && this.state.error.toString()} <br /> {this.state.errorInfo.componentStack} </details> )} <button onClick={() => window.location.reload()}> Refresh Page </button> </div> ); } return this.props.children; } } // Usage function App() { return ( <ErrorBoundary> <YourApp /> </ErrorBoundary> ); }

Global Error Handler

// Global error handler setup class GlobalErrorHandler { constructor() { this.errors = []; this.setupHandlers(); } setupHandlers() { // Catch unhandled promise rejections window.addEventListener('unhandledrejection', (event) => { console.error('Unhandled promise rejection:', event.reason); this.handleError({ type: 'unhandledRejection', message: event.reason.message || 'Promise rejection', stack: event.reason.stack }); }); // Catch global errors window.addEventListener('error', (event) => { console.error('Global error:', event.error); this.handleError({ type: 'globalError', message: event.message, filename: event.filename, lineno: event.lineno, colno: event.colno, stack: event.error?.stack }); }); } handleError(error) { this.errors.push({ ...error, timestamp: new Date().toISOString() }); // Send to error tracking service this.reportError(error); // Show user-friendly message this.showErrorNotification(error); } reportError(error) { // Send to backend or error tracking service fetch('/api/log-error', { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify(error) }).catch(err => console.error('Failed to report error:', err)); } showErrorNotification(error) { // Show toast notification or modal const notification = document.createElement('div'); notification.className = 'error-notification'; notification.textContent = 'An error occurred. Our team has been notified.'; document.body.appendChild(notification); setTimeout(() => notification.remove(), 5000); } } // Initialize const errorHandler = new GlobalErrorHandler();

1. File upload mein FormData ka kya role hai aur isse kaise use karte hain?
Easy ‚ñº

Answer:

FormData ek JavaScript object hai jo form data ko key-value pairs mein encode karta hai, especially file uploads ke liye.

Key Points:

  • Files ko binary format mein send karne ke liye use hota hai
  • Multiple files aur additional data ek saath bhej sakte hain
  • Automatically correct Content-Type header set karta hai
  • XMLHttpRequest ya Fetch API ke saath use hota hai
// Example const formData = new FormData(); formData.append('file', fileInput.files[0]); formData.append('username', 'john'); formData.append('description', 'Profile picture'); fetch('/api/upload', { method: 'POST', body: formData // No need to set Content-Type });
2. Drag and Drop API mein kaunse events important hain aur unka sequence kya hai?
Medium ‚ñº

Answer:

Drag and Drop Events Sequence:

  1. dragstart: Jab user dragging start karta hai
  2. drag: Dragging ke dauran continuously fire hota hai
  3. dragenter: Jab dragged item drop zone mein enter karta hai
  4. dragover: Jab dragged item drop zone ke upar hover karta hai
  5. dragleave: Jab dragged item drop zone se bahar jata hai
  6. drop: Jab item drop hota hai
  7. dragend: Jab drag operation complete hota hai

Important Points:

  • preventDefault() call karna zaroori hai dragover aur drop events mein
  • dataTransfer object se files access karte hain
  • Visual feedback ke liye CSS classes add/remove karein
3. Client-side validation aur server-side validation mein kya difference hai? Dono kyun zaroori hain?
Medium ‚ñº

Answer:

Aspect Client-Side Server-Side
Location Browser mein Server pe
Speed Instant feedback Network delay
Security Bypass ho sakta hai Secure, bypass nahi ho sakta
Purpose UX improvement Data integrity & security

Dono Kyun Zaroori Hain:

  • Client-side: Better UX, instant feedback, reduced server load
  • Server-side: Security, data integrity, business logic enforcement

  • Server-side validation final line of defense hai

6. File upload security considerations kya hain?
Hard ‚ñº

Answer:

File Upload Security Considerations:

  1. File Type Validation:
    • MIME type check karein (client aur server dono pe)
    • File extension validate karein
    • Magic bytes check karein (actual file content)
  2. File Size Limits:
    • Maximum file size enforce karein
    • DoS attacks prevent karne ke liye
  3. File Name Sanitization:
    • Special characters remove karein
    • Path traversal attacks prevent karein (../, etc.)
    • Unique file names generate karein
  4. Storage Security:
    • Files ko web root ke bahar store karein
    • Direct access prevent karein
    • Virus scanning implement karein
  5. Authentication & Authorization:
    • User authentication verify karein
    • Upload permissions check karein
    • Rate limiting implement karein
